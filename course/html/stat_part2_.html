def mean_confidence_interval(data, confidence=0.95):
    a = np.array(data)
    n = len(a)
    m, s = np.mean(a), np.std(a)
    h = s * stats.t.ppf((1 + confidence) / 2, n - 1) / np.sqrt(n)
    return m-h, m, m+h

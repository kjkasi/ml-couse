import math
def quantile(list_,a):
    k=(1/a)
    h=math.floor(len(list_)/k)
    if len(list_)%2!=0:
        q=list_[h]
    if len(list_)%2==0:
        q=(list_[h-1]+list_[h])/2
    return q
